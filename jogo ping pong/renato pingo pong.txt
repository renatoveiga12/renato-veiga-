    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        
        // Game state
        let gameMode = 'ai';
        let difficulty = 'easy';
        let isPaused = false;
        let isGameOver = false;
        let animationId;
        
        // Game objects
        const paddleWidth = 15;
        const paddleHeight = 80;
        const ballSize = 12;
        
        let player1 = {
            y: canvas.height / 2 - paddleHeight / 2,
            score: 0,
            speed: 0,
            maxSpeed: 6
        };
        
        let player2 = {
            y: canvas.height / 2 - paddleHeight / 2,
            score: 0,
            speed: 0,
            maxSpeed: 6
        };
        
        let ball = {
            x: canvas.width / 2,
            y: canvas.height / 2,
            dx: 5,
            dy: 3,
            speed: 6,
            trail: []
        };
        
        // AI settings
        const aiSettings = {
            easy: { speed: 3, reaction: 0.3, error: 40 },
            medium: { speed: 4.5, reaction: 0.6, error: 20 },
            hard: { speed: 5.5, reaction: 0.9, error: 5 }
        };
        
        // Input handling
        const keys = {};
        
        document.addEventListener('keydown', (e) => {
            keys[e.key.toLowerCase()] = true;
            if (e.key === 'Escape' && !document.getElementById('mainMenu').classList.contains('hidden')) {
                togglePause();
            }
        });
        
        document.addEventListener('keyup', (e) => {
            keys[e.key.toLowerCase()] = false;
        });
        
        // Touch controls for mobile
        let touchStartY = 0;
        canvas.addEventListener('touchstart', (e) => {
            touchStartY = e.touches[0].clientY;
        });
        
        canvas.addEventListener('touchmove', (e) => {
            e.preventDefault();
            const touchY = e.touches[0].clientY;
            const rect = canvas.getBoundingClientRect();
            const relativeY = touchY - rect.top;
            
            if (gameMode === 'pvp') {
                if (e.touches[0].clientX < rect.left + rect.width / 2) {
                    player1.y = relativeY - paddleHeight / 2;
                } else {
                    player2.y = relativeY - paddleHeight / 2;
                }
            } else {
                player1.y = relativeY - paddleHeight / 2;
            }
        });
        
        function setDifficulty(level) {
            difficulty = level;
            document.querySelectorAll('.diff-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');
        }
        
        function startGame(mode) {
            gameMode = mode;
            document.getElementById('mainMenu').classList.add('hidden');
            document.getElementById('player2Name').textContent = mode === 'ai' ? 'Computador' : 'Jogador 2';
            document.getElementById('p2Controls').style.opacity = mode === 'pvp' ? '1' : '0.5';
            resetGame();
            gameLoop();
        }
        
        function resetGame() {
            player1.y = canvas.height / 2 - paddleHeight / 2;
            player2.y = canvas.height / 2 - paddleHeight / 2;
            player1.score = 0;
            player2.score = 0;
            updateScore();
            resetBall();
            isGameOver = false;
            isPaused = false;
        }
        
        function resetBall() {
            ball.x = canvas.width / 2;
            ball.y = canvas.height / 2;
            ball.speed = 6;
            ball.dx = (Math.random() > 0.5 ? 1 : -1) * ball.speed;
            ball.dy = (Math.random() * 2 - 1) * ball.speed;
            ball.trail = [];
        }
        
        function updateScore() {
            document.getElementById('score1').textContent = player1.score;
            document.getElementById('score2').textContent = player2.score;
        }
        
        function update() {
            if (isPaused || isGameOver) return;
            
            // Player 1 controls (W/S)
            if (keys['w'] && player1.y > 0) {
                player1.y -= player1.maxSpeed;
            }
            if (keys['s'] && player1.y < canvas.height - paddleHeight) {
                player1.y += player1.maxSpeed;
            }
            
            // Player 2 controls or AI
            if (gameMode === 'pvp') {
                if (keys['arrowup'] && player2.y > 0) {
                    player2.y -= player2.maxSpeed;
                }
                if (keys['arrowdown'] && player2.y < canvas.height - paddleHeight) {
                    player2.y += player2.maxSpeed;
                }
            } else {
                updateAI();
            }
            
            // Ball trail
            ball.trail.push({ x: ball.x, y: ball.y });
            if (ball.trail.length > 10) ball.trail.shift();
            
            // Ball movement
            ball.x += ball.dx;
            ball.y += ball.dy;
            
            // Ball collision with top/bottom
            if (ball.y <= 0 || ball.y >= canvas.height) {
                ball.dy = -ball.dy;
                ball.y = ball.y <= 0 ? 0 : canvas.height;
            }
            
            // Ball collision with paddles
            if (ball.dx < 0 && 
                ball.x <= paddleWidth + 10 && 
                ball.x >= 10 && 
                ball.y >= player1.y && 
                ball.y <= player1.y + paddleHeight) {
                
                ball.dx = -ball.dx * 1.05;
                ball.speed *= 1.02;
                let hitPos = (ball.y - player1.y) / paddleHeight;
                ball.dy = 8 * (hitPos - 0.5);
                ball.x = paddleWidth + 10;
            }
            
            if (ball.dx > 0 && 
                ball.x >= canvas.width - paddleWidth - 10 && 
                ball.x <= canvas.width - 10 && 
                ball.y >= player2.y && 
                ball.y <= player2.y + paddleHeight) {
                
                ball.dx = -ball.dx * 1.05;
                ball.speed *= 1.02;
                let hitPos = (ball.y - player2.y) / paddleHeight;
                ball.dy = 8 * (hitPos - 0.5);
                ball.x = canvas.width - paddleWidth - 10;
            }
            
            // Scoring
            if (ball.x < 0) {
                player2.score++;
                updateScore();
                checkWin();
                resetBall();
            } else if (ball.x > canvas.width) {
                player1.score++;
                updateScore();
                checkWin();
                resetBall();
            }
        }
        
        function updateAI() {
            const ai = aiSettings[difficulty];
            const targetY = ball.y - paddleHeight / 2;
            const currentY = player2.y;
            
            // Add some error based on difficulty
            const error = (Math.random() - 0.5) * ai.error;
            const adjustedTarget = targetY + error;
            
            // Reaction delay
            if (Math.random() > ai.reaction) return;
            
            if (Math.abs(adjustedTarget - currentY) > ai.speed) {
                if (adjustedTarget > currentY) {
                    player2.y += ai.speed;
                } else {
                    player2.y -= ai.speed;
                }
            }
            
            // Keep paddle in bounds
            player2.y = Math.max(0, Math.min(canvas.height - paddleHeight, player2.y));
        }
        
        function checkWin() {
            if (player1.score >= 10 || player2.score >= 10) {
                isGameOver = true;
                const winner = player1.score >= 10 ? 'Jogador 1' : (gameMode === 'ai' ? 'Computador' : 'Jogador 2');
                document.getElementById('winnerText').textContent = winner + ' Venceu!';
                document.getElementById('gameOverMenu').classList.remove('hidden');
            }
        }
        
        function draw() {
            // Clear canvas with fade effect
            ctx.fillStyle = 'rgba(0, 0, 0, 0.3)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Draw center line
            ctx.strokeStyle = 'rgba(0, 245, 255, 0.2)';
            ctx.lineWidth = 2;
            ctx.setLineDash([10, 10]);
            ctx.beginPath();
            ctx.moveTo(canvas.width / 2, 0);
            ctx.lineTo(canvas.width / 2, canvas.height);
            ctx.stroke();
            ctx.setLineDash([]);
            
            // Draw ball trail
            ball.trail.forEach((pos, index) => {
                const alpha = index / ball.trail.length;
                ctx.beginPath();
                ctx.arc(pos.x, pos.y, ballSize * alpha * 0.8, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(0, 245, 255, ${alpha * 0.5})`;
                ctx.fill();
            });
            
            // Draw ball with glow
            ctx.shadowBlur = 20;
            ctx.shadowColor = '#00f5ff';
            ctx.beginPath();
            ctx.arc(ball.x, ball.y, ballSize, 0, Math.PI * 2);
            ctx.fillStyle = '#00f5ff';
            ctx.fill();
            ctx.shadowBlur = 0;
            
            // Draw paddles with glow
            ctx.shadowBlur = 15;
            
            // Player 1 paddle
            ctx.shadowColor = '#00f5ff';
            ctx.fillStyle = '#00f5ff';
            ctx.fillRect(10, player1.y, paddleWidth, paddleHeight);
            
            // Player 2 paddle
            ctx.shadowColor = '#ff00ff';
            ctx.fillStyle = '#ff00ff';
            ctx.fillRect(canvas.width - paddleWidth - 10, player2.y, paddleWidth, paddleHeight);
            
            ctx.shadowBlur = 0;
        }
        
        function gameLoop() {
            update();
            draw();
            animationId = requestAnimationFrame(gameLoop);
        }
        
        function togglePause() {
            if (document.getElementById('mainMenu').classList.contains('hidden') && 
                document.getElementById('gameOverMenu').classList.contains('hidden')) {
                isPaused = !isPaused;
                document.getElementById('pauseMenu').classList.toggle('hidden');
                document.getElementById('pauseBtn').textContent = isPaused ? 'Continuar' : 'Pausar';
            }
        }
        
        function resumeGame() {
            isPaused = false;
            document.getElementById('pauseMenu').classList.add('hidden');
            document.getElementById('pauseBtn').textContent = 'Pausar';
        }
        
        function restartGame() {
            document.getElementById('gameOverMenu').classList.add('hidden');
            resetGame();
        }
        
        function backToMenu() {
            isPaused = false;
            isGameOver = false;
            cancelAnimationFrame(animationId);
            document.getElementById('pauseMenu').classList.add('hidden');
            document.getElementById('gameOverMenu').classList.add('hidden');
            document.getElementById('mainMenu').classList.remove('hidden');
            document.getElementById('pauseBtn').textContent = 'Pausar';
            
            // Clear canvas
            ctx.fillStyle = '#0f0f23';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
        }
        
        // Initial draw
        ctx.fillStyle = '#0f0f23';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
    </script>
</body>
</html>'''
